สคริปต์ SQL ด้านล่างนี้ประกอบด้วยการสร้างตาราง (Tables), การตั้งค่าข้อจำกัด (Constraints), ความสัมพันธ์ (Foreign Keys) รวมถึงการเปิดใช้งานระบบความปลอดภัย Row Level Security (RLS) เพื่อปกป้องข้อมูลความลับของลูกความ

-- 1. ตารางคดีความ (cases)
CREATE TABLE cases (
    case_id TEXT PRIMARY KEY DEFAULT 'CASE-' || to_char(current_timestamp, 'YYYYMMDD-') || LPAD(nextval('cases_seq')::text, 4, '0'),
    black_number TEXT NOT NULL,
    red_number TEXT,
    court_name TEXT NOT NULL,
    case_type TEXT CHECK (case_type IN ('Civil', 'Criminal', 'Labor', 'Administrative', 'Other')),
    case_status TEXT DEFAULT 'Draft' CHECK (case_status IN ('Draft', 'In_Progress', 'Appealing', 'Closed')),
    claim_amount NUMERIC(15, 2) DEFAULT 0.00,
    description TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL
);

-- 2. ตารางผู้เกี่ยวข้องในคดี (contacts)
CREATE TABLE contacts (
    contact_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    case_id TEXT REFERENCES cases(case_id) ON DELETE CASCADE,
    full_name TEXT NOT NULL,
    role TEXT CHECK (role IN ('Client_Plaintiff', 'Client_Defendant', 'Opponent', 'Witness', 'Other')),
    citizen_id TEXT,
    contact_info TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL
);

-- 3. ตารางตารางนัดหมายและปฏิทินคดี (schedules)
CREATE TABLE schedules (
    event_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    case_id TEXT REFERENCES cases(case_id) ON DELETE CASCADE,
    event_date TIMESTAMP WITH TIME ZONE NOT NULL,
    event_type TEXT NOT NULL,
    remind_days INT DEFAULT 7,
    notes TEXT,
    is_completed BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL
);

-- 4. ตารางจัดเก็บเอกสารทางกฎหมาย (documents)
CREATE TABLE documents (
    doc_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    case_id TEXT REFERENCES cases(case_id) ON DELETE CASCADE,
    doc_name TEXT NOT NULL,
    file_path TEXT NOT NULL, -- ลิงก์ไปยัง Supabase Storage Bucket
    uploaded_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL
);

ข้อแนะนำด้านความปลอดภัยและการรักษาข้อมูลความลับ (Security & RLS)

เนื่องจากข้อมูลทางกฎหมายมีความอ่อนไหวสูงมาก คุณทนายความต้องเปิดใช้งาน Row Level Security (RLS) เพื่อป้องกันไม่ให้บุคคลภายนอกเข้าถึงข้อมูลได้โดยตรง:

-- เปิดใช้งาน RLS สำหรับทุกตาราง
ALTER TABLE cases ENABLE ROW LEVEL SECURITY;
ALTER TABLE contacts ENABLE ROW LEVEL SECURITY;
ALTER TABLE schedules ENABLE ROW LEVEL SECURITY;
ALTER TABLE documents ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Allow authenticated users access" ON cases
